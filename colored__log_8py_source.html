30def get_format(line_no: bool = False, threads: bool = False):
31 """!
32 @brief Get the configurable log format string
33 @param line_no : include line number in log string
34 @param threads : include thread name in log string
35 @return log format string
36 """
37 if not line_no and not threads:
38 msg_format = S_LOG_MSG_FORMAT_WO_LINENO_WO_THREADS
39 elif not line_no and threads:
40 msg_format = S_LOG_MSG_FORMAT_WO_LINENO_W_THREADS

67class ColorFormatter(logging.Formatter):
68 """!
69 @brief Logging formatter that colors logging messages depending on their level
70 @param fmt : [optional] overwrite format with a complete custom log format -> ignores all other parameters
71 @param line_no : [optional] include line number in log format string
72 @param threads : [optional] include thread names in log format string
73 @param data_format : [optional] date format
74 """
75 fg_grey = "\033[90m"
76 fg_yellow = "\033[33m"
77 fg_red = "\033[31m"
78 bg_red = "\033[41m"
79 reset = "\033[0m"

81 def __init__(self, fmt: Optional[str] = None, line_no: bool = True, threads: bool = False, data_format: str = None, **kwargs) -> None:
82 super().__init__(fmt, **kwargs)
83 msg_format = fmt if fmt else get_format(line_no=line_no, threads=threads)
84 self.d_formats = {
85 logging.DEBUG: self.fg_grey + msg_format + self.reset,
86 logging.INFO: self.reset + msg_format,
87 logging.WARNING: self.fg_yellow + msg_format + self.reset,
88 logging.ERROR: self.fg_red + msg_format + self.reset,
89 logging.CRITICAL: self.bg_red + msg_format + self.reset
90 }
91 self.data_format = data_format
92
93 def format(self, record):
94 """!
95 @brief Overwrite format method of logging.Formatter to use colors formatting
96 @param record : log record
97 @return formatted log record
98 """
99 log_fmt = self.d_formats.get(record.levelno)
100 formatter = logging.Formatter(fmt=log_fmt, datefmt=self.data_format)
101 return formatter.format(record)
102
103 def formatException(self, ei):
104 """!
105 @brief Overwrite formatException method of logging. Formatter to use critical color formatting
106 @param ei : exception info
107 @return formatted exception log record
108 """
109 log_fmt = self.d_formats.get(logging.CRITICAL)
110 formatter = logging.Formatter(log_fmt, datefmt=self.data_format)
111 record = super().formatException(ei)
112 return formatter.format(record)
