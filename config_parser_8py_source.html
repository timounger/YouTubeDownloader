31 def load_configuration(self, doxyfile: str) -> dict:
32 """!
33 @brief Parse a Doxygen configuration file
34 @param doxyfile : doxyfile: Path to the Doxygen configuration file
35 @return A dict with all doxygen configuration
36 """
37
38 if not os.path.exists(doxyfile):
39 logging.error("Impossible to access to %s", doxyfile)
40 raise FileNotFoundError(doxyfile)
41
42 configuration: dict[str, str | list] = {}
43
44 with open(doxyfile, 'r', encoding='utf-8') as file:
45
46 in_multiline_option = False
47 current_multiline_option_name = None
48
49 for line in file.readlines():
50 line = line.strip()
51 if len(line) == 0:
52 continue
53
54 if self.__is_comment_line(line):
55 continue
56
57 if in_multiline_option:
58 if not line.endswith('\\'):
59 in_multiline_option = False
60 option_value = line.rstrip('\\').strip()
61 if current_multiline_option_name is not None:
62 configuration[current_multiline_option_name].append(option_value)
63
64 elif self.__is_first_line_of_multiline_option(line):
65 current_multiline_option_name, option_value = self.__extract_multiline_option_name_and_first_value(line)
66 configuration[current_multiline_option_name] = [option_value]
67 in_multiline_option = True
68
69 elif self.__is_single_line_option(line):
70 option_name, option_value = self.__extract_single_line_option_name_and_value(line)
71 configuration[option_name] = option_value
72
73 return configuration

75 def store_configuration(self, config: dict, doxyfile: str):
76 """!
77 @brief Store the doxygen configuration to the disk
78 @param config : The doxygen configuration you want to write on disk
79 @param doxyfile : The output path where configuration will be written. If the file exist, it will be truncated
80 """
81
82 logging.debug("Store configuration in %s", doxyfile)
83
84 lines = []
85 for option_name, option_value in config.items():
86 if isinstance(option_value, list):
87 # Changes by Timo Unger (add with force_double_quote) */
88 lines.append(f"{option_name} = {self.__add_double_quote_if_required(option_value[0], force_double_quote=True)} \\")
89 lines.extend([f"\t{self.__add_double_quote_if_required(value, force_double_quote=True)} \\" for value in option_value[1:-1]])
90 lines.append(f"\t{self.__add_double_quote_if_required(option_value[-1], force_double_quote=True)}")
91 elif isinstance(option_value, str):
92 lines.append(f"{option_name} = {self.__add_double_quote_if_required(option_value)}")
93
94 with open(doxyfile, 'w', encoding='utf-8') as file:
95 file.write("\n".join(lines))

97 def __extract_multiline_option_name_and_first_value(self, line: str) -> Tuple[str, str]:
98 """!
99 @brief Extract the option name and the first value of multi line option
100 @param line : The line you want to parse
101 @return the option name and the option first value
102 """
103
104 matches = self.__first_line_of_multine_option_regex.search(line)
105 if matches is None or len(matches.groups()) != 2:
106 logging.error("Impossible to extract first value off multi line option from: %s", line)
107 raise ParseException(f"Impossible to extract first value off multi line option from: {line}")
108
109 return matches.group(1), self.__remove_double_quote_if_required(matches.group(2))
