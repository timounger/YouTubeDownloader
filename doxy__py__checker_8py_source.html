88 def check_function(self, func_def: ast.FunctionDef, class_docstring: str = None) -> List[str]:
89 """!
90 @brief Check function
91 @param func_def : function definition
92 @param class_docstring : docstring of class
93 @return list of findings in function
94 """
95 findings = []
96 documented_params = set()
97 docstring = ast.get_docstring(func_def) or class_docstring
98 if docstring:
99 # Get documented parameters and check for duplicate documentation
100 for line in docstring.splitlines():
101 if line.lstrip().startswith(PARAM_DOC_PREFIX):
102 param_name = line.split()[1].rstrip(":")
103 if param_name in documented_params:
104 findings.append(f"{param_name} is documented multiple times")
105 else:
106 documented_params.add(param_name)
107
108 # Check that all functional parameters are documented
109 for arg in func_def.args.args:
110 if arg.arg not in L_IGNORE_PARAM:
111 if arg.arg not in documented_params:
112 findings.append(f"{arg.arg} is not documented")
113 if CHECK_TYPING and arg.annotation is None:
114 findings.append(f"{arg.arg} has no typing")
115
116 # Check that the documented parameters are present
117 for documented_param in documented_params:
118 all_args = [arg.arg for arg in func_def.args.args]
119 if func_def.args.vararg:
120 all_args.append(func_def.args.vararg.arg)
121 if func_def.args.kwarg:
122 all_args.append(func_def.args.kwarg.arg)
123 if documented_param not in all_args:
124 findings.append(f"{documented_param} is documented but not used")
125
126 # Check that the function return is correctly specified: True: need doc, False no doc, None: optional doc
127 doc_has_return = RETURN_DOC_PREFIX in docstring
128 func_has_return = False
129 for node in ast.walk(func_def):
130 if isinstance(node, ast.Return) and node.value is not None: # function has no return None
131 func_has_return = True
132 break
133 if isinstance(node, ast.Call) and isinstance(node.func, ast.Attribute) and node.func.attr == "exit":
134 module = node.func.value
135 if isinstance(module, ast.Name) and module.id == "sys":
136 func_has_return = None # documentation of "sys.exit" is optional
137 if func_has_return is not None:
138 if doc_has_return != func_has_return:
139 if func_has_return:
140 findings.append("Return type is not documented")
141 else:
142 findings.append("Return type is documented but not used")
143 if CHECK_TYPING and func_has_return:
144 return_annotation = astunparse.unparse(func_def.returns).strip() if func_def.returns else None
145 if return_annotation is None:
146 findings.append("Return type has no typing")
147
148 return findings

150 def check_file(self, file_path: str) -> List[str]:
151 """!
152 @brief Check file for missing parameter description
153 @param file_path : file name
154 @return list of findings in file
155 """
156 with open(file_path, mode='r', encoding='utf-8') as file:
157 code = file.read()
158
159 tree = ast.parse(code)
160
161 file_findings = []
162 for node in ast.walk(tree):
163 func_def = None
164 check_findings = []
165 if isinstance(node, ast.ClassDef):
166 class_docstring = ast.get_docstring(node)
167 for subnode in node.body:
168 if isinstance(subnode, (ast.FunctionDef, ast.AsyncFunctionDef)):
169 if subnode.name == INIT_FUNCTION:
170 func_def = subnode
171 check_findings = self.check_function(func_def, class_docstring)
172 elif isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
173 func_def = node
174 check_findings = self.check_function(func_def)
175
176 # print findings
177 function_finding = ""
178 if func_def and check_findings:
179 function_finding = f"{file_path}:{func_def.lineno} {func_def.name}"
180 for warning in check_findings:
181 function_finding += f"\n {warning}"
182
183 if function_finding:
184 file_findings.append(function_finding)
185
186 return file_findings
