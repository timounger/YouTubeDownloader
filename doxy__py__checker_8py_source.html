95 def get_doc_params(self, docstring: str, findings: list) -> set:
96 """!
97 @brief Get documented parameters from docstring
98 @param docstring : docstring of class
99 @param findings : list to add doc warnings
100 @return collection with documented parameters
101 """
102 documented_params = set()
103 for line in docstring.splitlines():
104 if line.lstrip().startswith(PARAM_DOC_PREFIX):
105 l_param_name = line.split()
106 if len(l_param_name) >= 2: # minimum two names: @param, param_name
107 param_name = l_param_name[1].rstrip(":")
108 if param_name in documented_params:
109 findings.append(f"{param_name} is documented multiple times")
110 else:
111 documented_params.add(param_name)
112 else:
113 findings.append(f"No parameter name found. Line content '{line}'")
114 return documented_params

116 def check_return(self, func_def: ast.FunctionDef | ast.AsyncFunctionDef, docstring: str) -> list[str]:
117 """!
118 @brief Check for documented return value
119 @param func_def : function definition
120 @param docstring : docstring of function
121 @return list of return findings in function
122 """
123 findings = []
124 doc_has_return = RETURN_DOC_PREFIX in docstring
125 func_has_return: bool | None = False # True: need doc, False: no doc, None: optional doc
126 for node in ast.walk(func_def):
127 if isinstance(node, ast.Return) and node.value is not None: # function has no return None
128 func_has_return = True
129 break
130 if isinstance(node, ast.Call) and isinstance(node.func, ast.Attribute) and (node.func.attr == "exit"):
131 module = node.func.value
132 if isinstance(module, ast.Name) and (module.id == "sys"):
133 func_has_return = None # documentation of "sys.exit" is optional
134 if func_has_return is not None:
135 if doc_has_return != func_has_return:
136 if func_has_return:
137 findings.append("Return type is not documented")
138 else:
139 findings.append("Return type is documented but not used")
140 if CHECK_TYPING and func_has_return:
141 return_annotation = astunparse.unparse(func_def.returns).strip() if func_def.returns else None
142 if return_annotation is None:
143 findings.append("Return type has no typing")
144 return findings

146 def check_function(self, func_def: ast.FunctionDef | ast.AsyncFunctionDef, class_docstring: Optional[str] = None) -> list[str]:
147 """!
148 @brief Check function
149 @param func_def : function definition
150 @param class_docstring : docstring of class
151 @return list of findings in function
152 """
153 findings: list[str] = []
154 docstring = ast.get_docstring(func_def) or class_docstring
155 if docstring:
156 # Get documented parameters and check for duplicate documentation
157 documented_params = self.get_doc_params(docstring, findings)
158
159 # Check that all functional parameters are documented
160 for arg in func_def.args.args:
161 if arg.arg not in L_IGNORE_PARAM:
162 if arg.arg not in documented_params:
163 findings.append(f"{arg.arg} is not documented")
164 if CHECK_TYPING and arg.annotation is None:
165 findings.append(f"{arg.arg} has no typing")
166
167 # Check that the documented parameters are present
168 for documented_param in documented_params:
169 all_args = [arg.arg for arg in func_def.args.args]
170 if func_def.args.vararg:
171 all_args.append(func_def.args.vararg.arg)
172 if func_def.args.kwarg:
173 all_args.append(func_def.args.kwarg.arg)
174 if documented_param not in all_args:
175 findings.append(f"{documented_param} is documented but not used")
176
177 # Check that the function return is correctly specified
178 findings += self.check_return(func_def, docstring)
179
180 return findings

182 def check_file(self, file_path: str) -> list[str]:
183 """!
184 @brief Check file for missing parameter description
185 @param file_path : file name
186 @return list of findings in file
187 """
188 with open(file_path, mode="r", encoding="utf-8") as file:
189 code = file.read()
190
191 tree = ast.parse(code)
192
193 file_findings = []
194 for node in ast.walk(tree):
195 func_def = None
196 check_findings = []
197 if isinstance(node, ast.ClassDef):
198 class_docstring = ast.get_docstring(node)
199 for subnode in node.body:
200 if isinstance(subnode, (ast.FunctionDef, ast.AsyncFunctionDef)):
201 if subnode.name == INIT_FUNCTION:
202 func_def = subnode
203 check_findings = self.check_function(func_def, class_docstring)
204 elif isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
205 func_def = node
206 check_findings = self.check_function(func_def)
207
208 # print findings
209 function_finding = ""
210 if func_def and check_findings:
211 function_finding = f"{file_path}:{func_def.lineno} {func_def.name}"
212 for warning in check_findings:
213 function_finding += f"\n {warning}"
214
215 if function_finding:
216 file_findings.append(function_finding)
