408 def set_configuration(self, s_type: str, value: Any, b_override: bool = True):
409 """!
410 @brief Set doxygen configuration.
411 @param s_type : type to set in configuration
412 @param value : value to set for s_type in configuration
413 @param b_override : info if selected default setting should override
414 """
415 if b_override or (self.get_configuration(s_type) in L_OVERRIDE):
416 if s_type in self.d_settings:
417 if isinstance(value, list):
418 if self.d_settings[s_type] in L_OVERRIDE:
419 self.d_settings[s_type] = []
420 self.d_settings[s_type].extend(value)
421 else:
422 self.d_settings[s_type] = value
423 else:
424 self.l_warnings.append(f"{s_type} not in configuration to set")
425
426 def get_configuration(self, s_type: str) -> Any:
427 """!
428 @brief Get type in doxygen configuration.
429 @param s_type : type in configuration get setting
430 @return configuration settings
431 """
432 if s_type in self.d_settings:
433 value = self.d_settings[s_type]
434 else:
435 value = "Invalid"
436 self.l_warnings.append(f"{s_type} not in configuration to get")
437 return value

439 def prepare_doxyfile_configuration(self):
440 """!
441 @brief Prepare doxfile configuration with default settings that depend on other parameters.

459 b_is_version = True
460 for s_char in s_version:
