123 def progress_callback(self, _stream, _chunk, bytes_remaining):
124 """!
125 @brief Calculate process and update process bar
126 @param _stream : stream
127 @param _chunk : chunk
128 @param bytes_remaining : bytes remaining
129 """
130 if not self.b_first_callback_call:
131 self.i_file_size = bytes_remaining
132 self.i_last_bytes_remaining = bytes_remaining
133 self.b_first_callback_call = True
134 else:
135 f_actual_time = time.time()
136 f_past_time = f_actual_time - self.f_time_stamp
137 if self.f_time_stamp != 0: # set calculate time only for second call
138 i_actual_speed = int((self.i_last_bytes_remaining - bytes_remaining) / f_past_time)
139 i_history_len = len(self.l_speed_history)
140 if i_history_len < I_SPEED_AVERAGE_VALUES:
141 self.l_speed_history.append(i_actual_speed)
142 else:
143 for i, value in enumerate(self.l_speed_history):
144 if i != 0:
145 self.l_speed_history[i - 1] = value
146 self.l_speed_history[I_SPEED_AVERAGE_VALUES - 1] = i_actual_speed
147 i_average_speed = statistics.mean(self.l_speed_history)
148 i_remaining_seconds = int(bytes_remaining / i_average_speed)
149 self.main_controller.o_status.config(text=f'Download läuft... noch {i_remaining_seconds}sek', fg="blue")
150 self.f_time_stamp = f_actual_time
151 self.i_last_bytes_remaining = bytes_remaining
152 i_percent = int(((self.i_file_size - bytes_remaining) / self.i_file_size) * 100)
153 i_percent_diff = i_percent - self.i_last_percent
154 for _ in range(i_percent_diff):
155 self.main_controller.o_progress.step()
156 self.main_controller.style.configure('text.Horizontal.TProgressbar', text=f'{i_percent}%')
157 self.i_last_percent = i_percent
