46def patched_get_throttling_function_name(js: str) -> str:
47 """!
48 @brief Patched get thottling function name
49 @param js : data
50 @return throttling
51 """
52 function_patterns = [
53 r'a\.[a-zA-Z]\s*&&\s*\‍([a-z]\s*=\s*a\.get\‍("n"\‍)\‍)\s*&&.*?\|\|\s*([a-z]+)',
54 r'\‍([a-z]\s*=\s*([a-zA-Z0-9$]+)(\[\d+\])?\‍([a-z]\‍)',
55 r'\‍([a-z]\s*=\s*([a-zA-Z0-9$]+)(\[\d+\])\‍([a-z]\‍)',
56 ]
57 for pattern in function_patterns:
58 regex = re.compile(pattern)
59 function_match = regex.search(js)
60 if function_match:
61 if len(function_match.groups()) == 1:
62 return function_match.group(1)
63 idx = function_match.group(2)
64 if idx:
65 idx = idx.strip("[]")
66 array = re.search(
67 r'var {nfunc}\s*=\s*(\[.+?\]);'.format(
68 nfunc=re.escape(function_match.group(1))),
69 js
70 )
71 if array:
72 array = array.group(1).strip("[]").split(",")
73 array = [x.strip() for x in array]
74 return array[int(idx)]
75
76 raise RegexMatchError(
77 caller="get_throttling_function_name", pattern="multiple"
78 )

181 def progress_callback(self, _stream: Stream, _chunk: bytes, bytes_remaining: int) -> None:
182 """!
183 @brief Calculate process and update process bar
184 @param _stream : stream
185 @param _chunk : chunk
186 @param bytes_remaining : bytes remaining

193 f_actual_time = time.time()
194 f_past_time = f_actual_time - self.f_time_stamp
195 if self.f_time_stamp != 0: # set calculate time only for second call

197 i_history_len = len(self.l_speed_history)
198 if i_history_len < I_SPEED_AVERAGE_VALUES:
199 self.l_speed_history.append(i_actual_speed)

201 for i, value in enumerate(self.l_speed_history):
202 if i != 0:
203 self.l_speed_history[i - 1] = value
204 self.l_speed_history[I_SPEED_AVERAGE_VALUES - 1] = i_actual_speed
205 i_average_speed = statistics.mean(self.l_speed_history)
206 i_remaining_seconds = int(bytes_remaining / i_average_speed)

208 self.f_time_stamp = f_actual_time
