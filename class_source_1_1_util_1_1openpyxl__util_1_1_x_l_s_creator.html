103 def set_column_autowidth(self, worksheet: Worksheet, limited: bool = True) -> None:
104 """!
105 @brief Set automatic column width of worksheet.
106 @param worksheet : target worksheet to modify
107 @param limited : True: limit to MAX_COLUMN_WIDTH, False: no limit
108 """
109 for i, col_cells in enumerate(worksheet.columns, start=1):
110 max_col_len = 0
111 for j, cell in enumerate(col_cells):
112 if limited:
113 if j != 0 and cell.value is not None: # do not use first line description
114 max_col_len = max(max_col_len, len(str(cell.value).split("\n", maxsplit=1)[0]))
115 else:
116 for line in str(cell.value).split("\n") if cell.value is not None else []:
117 max_col_len = max(max_col_len, len(line))
118 if limited:
119 max_col_len = min(max_col_len, MAX_COLUMN_WIDTH)
120 if max_col_len == 0:
121 worksheet.column_dimensions[get_column_letter(i)].hidden = True # hide empty columns
122 else:
123 worksheet.column_dimensions[get_column_letter(i)].width = (max_col_len + 1) * 0.10 * self.font_size
