13log = logging.getLogger("CheckIncludedPackages")

15# List of third party packages that may be contained in the PyInstaller executable.
16# Has to be manually extended if a new package gets added to the tool.
17L_ALLOWED_THIRD_PARTY_PACKAGES = [
18 # not possible to exclude
19 "PyInstaller",
20 # clipboard
21 "clipboard",
22 "pyperclip",
23 # youtube downloader
24 "pytube"
25]
26
27APP_NAME = "YouTubeDownloader"
28
29S_RELATIVE_PATH = fr"build\{APP_NAME}\xref-{APP_NAME}.html"
30
31
32def check_included_packages() -> list:
33 """!
34 @brief Check included packages
35 @return status if included packages are okay
36 """
37 d_third_party = {}
38 d_buildin = {}
39 d_own_packs = {}
40
41 # Regular Expressions
42 regex_third_party = re.compile(r".env\/lib\/site-packages.*.py", re.IGNORECASE)
43 regex_third_party_name = re.compile(r"(?<=site-packages\/).*", re.IGNORECASE)
44 regex_builtin = re.compile(r".env\/lib\/.*.py", re.IGNORECASE)
45 regex_builtin_name = re.compile(r"(?<=lib\/).*", re.IGNORECASE)
46 regex_own_packs = re.compile(fr"{APP_NAME}\/.*")
47
48 # read PyInstaller modulegraph cross reference HTML
49 with open(S_RELATIVE_PATH, mode="r", encoding="utf-8") as o_html_file:
50 soup = BeautifulSoup(o_html_file, "html.parser")
51
52 # parse HTML
53 l_nodes = soup.find_all("div", class_="node")
54 for o_node in l_nodes:
55 l_targets = o_node.find_all("a", target="code")
56 for o_target in l_targets:
57 s_package_path = o_target["href"]
58 if regex_third_party.search(s_package_path):
59 matches = regex_third_party_name.search(s_package_path)
60 if matches is not None:
61 d_third_party[matches.group().split("/", maxsplit=1)[0].replace(".py", "")] = ""
62 elif regex_builtin.search(s_package_path):
63 matches = regex_builtin_name.search(s_package_path)
64 if matches is not None:
65 d_buildin[matches.group().replace(".py", "")] = ""
66 elif regex_own_packs.search(s_package_path):
67 matches = regex_own_packs.search(s_package_path)
68 if matches is not None:
69 d_own_packs[matches.group()] = ""
70
71 # print included packages
72 log.info("\nThird party packages:")
73 log.info("\n".join(list(d_third_party.keys())))
74 # log.info("\nOwn packages:")
75 # log.info("\n".join(list(d_own_packs.keys())))
76 # log.info("\nPython buildin packages:")
77 # log.info("\n".join(list(d_buildin.keys())))
78
79 # check third party packages
80 l_not_allowed_packages = []
81 for s_package in list(d_third_party.keys()):
82 if s_package not in L_ALLOWED_THIRD_PARTY_PACKAGES:
83 l_not_allowed_packages.append(f"ERROR PyInstaller included an unknown package in the executable: \'{s_package}\'")
84 return l_not_allowed_packages
