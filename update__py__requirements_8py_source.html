47def set_column_autowidth(worksheet: Worksheet, b_limit: bool = True):
48 """!
49 @brief Set automatic column width of worksheet.
50 @param worksheet : select worksheet
51 @param b_limit : status if width has a max limit
52 """
53 for i, col_cells in enumerate(worksheet.columns, start=1):
54 i_max_col_len = 0
55 for j, cell in enumerate(col_cells):
56 if b_limit:
57 if j != 0: # do use not use first line description
58 i_max_col_len = max(i_max_col_len, len(str(cell.value).split("n", maxsplit=1)[0]))
59 else:
60 for s_line in str(cell.value).split("/n"):
61 i_max_col_len = max(i_max_col_len, len(s_line))
62 if b_limit:
63 i_max_col_len = min(i_max_col_len, MAX_COLUMN_WIDTH)
64 if i_max_col_len == 0:
65 worksheet.column_dimensions[get_column_letter(i)].hidden = True # hide empty lines
66 else:
67 worksheet.column_dimensions[get_column_letter(i)].width = (i_max_col_len + 1) * 0.10 * FONT_SIZE

146def get_package_info(package: str) -> PackageInfo:
147 """!
148 @brief Upgrade package from text file to latest version
149 @param package : check latest version of this package
150 @return package info
151 """
152 try:
153 url = f"https://pypi.org/pypi/{package}/json"
154 response = requests.get(url, timeout=I_TIMEOUT)
155 response.raise_for_status()
156 package_info = response.json()
157 except requests.exceptions.RequestException as e:
158 print(f"Error occurred: {e}")
159 package_info = None
160 else:
161 l_requires_dist = []
162 requires_dist = package_info["info"]["requires_dist"]
163 if requires_dist:
164 for req in requires_dist:
165 l_requires_dist.append(req)
166 package_info = PackageInfo(package,
167 package_info["info"]["version"],
168 package_info["info"]["author"],
169 package_info["info"]["author_email"],
170 package_info["info"]["license"],
171 package_info["info"]["home_page"],
172 package_info["info"]["package_url"],
173 l_requires_dist,
174 package_info["info"]["requires_python"])
175 return package_info

178def update_packages(filename: str) -> List:
179 """!
180 @brief Upgrade package from text file to latest version
181 @param filename : file name
182 @return list with updated packages
183 """
184 with open(filename, mode="r", encoding="utf-8") as file:
185 lines = file.readlines()
186
187 i_update_cnt = 0
188 l_package_info = []
189 updated_lines = []
190 for line in lines:
191 if re.match(r"^\s*#", line): # comment line
192 updated_lines.append(line)
193 elif re.match(r"^\s*([\w\-]+)==([\w\.\-]+)\s*(#.*)?$", line): # check for fix version 'package==version'
194 package, version, comment = re.findall(r"^\s*([\w\-]+)==([\w\.\-]+)\s*(#.*)?$", line)[0]
195 if comment != "":
196 comment = f" {comment}"
197 package_info = get_package_info(package)
198 l_package_info.append(package_info)
199 if (package not in L_IGNORE_PACKAGES) and package_info.version and (package_info.version != version):
200 updated_lines.append(f"{package}=={package_info.version}{comment}\n")
201 i_update_cnt += 1
202 if B_PRINT_PACAKGES or B_PRINT_UPDATES_PACKAGES:
203 print(f"Updated: {package} {package_info.version}")
204 else:
205 updated_lines.append(line)
206 if B_PRINT_PACAKGES:
207 print(line)
208 else: # package without version
209 updated_lines.append(line)
210 if line.strip():
211 package_info = get_package_info(line)
212 l_package_info.append(package_info)
213 if B_PRINT_PACAKGES:
214 print(line)
215
216 with open(filename, mode="w", encoding="utf-8") as file:
217 file.writelines(updated_lines)
218 print(f"{i_update_cnt} packages updates in {filename}\n")
219 return l_package_info
