54def set_column_autowidth(worksheet: Worksheet, b_limit: bool = True):
55 """!
56 @brief Set automatic column width of worksheet.
57 @param worksheet : select worksheet
58 @param b_limit : status if width has a max limit
59 """
60 for i, col_cells in enumerate(worksheet.columns, start=1):
61 i_max_col_len = 0
62 for j, cell in enumerate(col_cells):
63 if b_limit:
64 if j != 0: # do use not use first line description
65 i_max_col_len = max(i_max_col_len, len(str(cell.value).split("n", maxsplit=1)[0]))
66 else:
67 for s_line in str(cell.value).split("/n"):
68 i_max_col_len = max(i_max_col_len, len(s_line))
69 if b_limit:
70 i_max_col_len = min(i_max_col_len, MAX_COLUMN_WIDTH)
71 if i_max_col_len == 0:
72 worksheet.column_dimensions[get_column_letter(i)].hidden = True # hide empty lines
73 else:
74 worksheet.column_dimensions[get_column_letter(i)].width = (i_max_col_len + 1) * 0.10 * FONT_SIZE

153def get_package_info(package: str) -> PackageInfo:
154 """!
155 @brief Upgrade package from text file to latest version
156 @param package : check latest version of this package
157 @return package info
158 """
159 try:
160 url = f"https://pypi.org/pypi/{package}/json"
161 response = requests.get(url, timeout=I_TIMEOUT)
162 response.raise_for_status()
163 package_info = response.json()
164 except requests.exceptions.RequestException as e:
165 log.error("Error occurred: %s", e)
166 package_info = None
167 else:
168 l_requires_dist = []
169 requires_dist = package_info["info"]["requires_dist"]
170 if requires_dist:
171 for req in requires_dist:
172 l_requires_dist.append(req)
173 package_info = PackageInfo(package,
174 package_info["info"]["version"],
175 package_info["info"]["author"],
176 package_info["info"]["author_email"],
177 package_info["info"]["license"],
178 package_info["info"]["home_page"],
179 package_info["info"]["package_url"],
180 l_requires_dist,
181 package_info["info"]["requires_python"])
182 return package_info

185def update_packages(filename: str) -> list:
186 """!
187 @brief Update package from text file to latest version
188 @param filename : file name
189 @return list with updated packages
190 """
191 with open(filename, mode="r", encoding="utf-8") as file:
192 lines = file.readlines()
193
194 i_update_cnt = 0
195 l_package_info = []
196 updated_lines = []
197 for line in lines:
198 if re.match(r"^\s*#", line): # comment line
199 updated_lines.append(line)
200 elif re.match(r"^\s*([\w\-]+)==([\w\.\-]+)\s*(#.*)?$", line): # check for fix version 'package==version'
201 package, version, comment = re.findall(r"^\s*([\w\-]+)==([\w\.\-]+)\s*(#.*)?$", line)[0]
202 if comment != "":
203 comment = f" {comment}"
204 package_info = get_package_info(package)
205 if (package_info is not None) and (package not in L_IGNORE_PACKAGES):
206 if package_info.version and (package_info.version != version):
207 updated_lines.append(f"{package}=={package_info.version}{comment}\n")
208 i_update_cnt += 1
209 log.info("Updated: %s %s", package, package_info.version)
210 else:
211 updated_lines.append(line)
212 log.debug(line)
213 l_package_info.append(package_info)
214 else:
215 updated_lines.append(line)
216 log.debug(line)
217 else: # package without version
218 updated_lines.append(line)
219 if line.strip():
220 package_info = get_package_info(line)
221 if package_info is not None:
222 l_package_info.append(package_info)
223 log.debug(line)
224
225 with open(filename, mode="w", encoding="utf-8") as file:
226 file.writelines(updated_lines)
227 log.info("%s packages updates in %s\n", i_update_cnt, filename)
228 return l_package_info
