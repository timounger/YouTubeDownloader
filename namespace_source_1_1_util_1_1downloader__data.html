29def resource_path(s_relative_path: str) -> str:
30 """!
31 @brief Returns the absolute path to a resource given by a relative path depending on the environment (EXE or Python)
32 @param s_relative_path : the relative path to a file or directory
33 @return absolute path to the resource

35 try:
36 # PyInstaller creates a temp folder and stores path in _MEIPASS

38 except Exception:
