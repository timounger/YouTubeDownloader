39def delete_selected_text(o_url):
40 """!
41 @brief Delete selected text
42 @param o_url : url
43 """
44 try:
45 s_select_text = o_url.selection_get()
46 except BaseException: # pylint: disable=bare-except
47 pass
48 else:
49 s_enty_text = o_url.get()
50 i_selected_text_length = len(s_select_text)
51 i_curser_pos = o_url.index('insert')
52 i_curser_pos_end = i_curser_pos + i_selected_text_length
53 s_text_to_check = s_enty_text[i_curser_pos: i_curser_pos_end]
54 if s_text_to_check == s_select_text:
55 o_url.delete(i_curser_pos, i_curser_pos_end)
56 else:
57 o_url.delete(i_curser_pos - i_selected_text_length, i_curser_pos)
